86def write_current_bus(transport_data): 

87 """ 

88 Function to write the results from an API call to gtfs transport for Ireland to our db 

89 Each trip update for each respective trip will be made into a CurrentBus object and then saved into a respective row 

90 in the database table. The table gets cleared everytime the function is run before saving new info. 

91 """ 

92 # Truncate table 

93 models.CurrentBus.objects.all().delete() 

94 

95 # Create list where we will store all our instances of CurrentBus that are generated in the for loop 

96 entries = [] 

97 

98 # Timestamp is the same for all items in the json object so we can create this and dt outside the loop 

99 timestamp = transport_data['header']['timestamp'] 

100 dt = datetime.now(tz=timezone.utc) 

101 

102 base = settings.BASE_DIR 

103 file_location = os.path.join(base, "dublinBus", "static", "dublinBus", "Dublin_bus_info", "json_files", "routes.json") 

104 with open(file_location) as routes: 

105 route_obj = json.load(routes) 

106 routes.close() 

107 route_list = [] 

108 for i in route_obj: 

109 route_list.append(route_obj[i]['route_id']) 

110 

111 # loop through all the entries in 'entity' and create a CurrentBus object for each one before saving to db 

112 for i in transport_data['entity']: 

113 trip = i["trip_update"]["trip"] 

114 if trip["route_id"] in route_list: 

115 temp_id = i["id"], 

116 temp_route_id = trip["route_id"], 

117 temp_schedule = trip["schedule_relationship"], 

118 temp_start_t = trip["start_time"], 

119 temp_start_d = trip["start_date"], 

120 # Now loop through all trip updates within the current trip 

121 try: 

122 for j in i["trip_update"]["stop_time_update"]: 

123 temp_stop_id = j["stop_id"] 

124 try: 

125 test = j["departure"] 

126 temp_delay = test["delay"] 

127 except: 

128 try: 

129 test = j['arrival'] 

130 temp_delay = test["delay"] 

131 except: 

132 temp_delay = 0 

133 except: 

134 temp_stop_id = None 

135 temp_delay = None 

136 

137 # Regardless of what happened in above try/except blocks we want to insert the row by creating the instance regardlesss 

138 finally: 

139 # Create one instance CurrentBus for each nested for loop 

140 latestUpdate = models.CurrentBus( 

141 timestamp=timestamp, 

142 dt=dt, 

143 trip_id=temp_id, 

144 route_id=temp_route_id, 

145 schedule=temp_schedule, 

146 start_t=temp_start_t, 

147 start_d=temp_start_d, 

148 stop_id=temp_stop_id, 

149 delay=temp_delay, 

150 ) 

151 # Now append this instance to our list of entries 

152 entries.append(latestUpdate) 

153 else: 

154 pass 

155 # Save all our entries to the database 

156 models.CurrentBus.objects.bulk_create(entries)
